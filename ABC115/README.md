### A-Christmas Eve Eve Eve
```Swift
func christmasEveEveEve(D: Int) -> String {
  var answer: String = ""

  switch D {
    case 25:
      answer = "Christmas"
    case 24:
      answer = "Christmas Eve"
    case 23:
      answer = "Christmas Eve Eve"
    case 22:
      answer = "Christmas Eve Eve Eve"
    default:
      answer = "Not Christmas"
  }

  return answer
}

print(christmasEveEveEve(D: 14))
```
