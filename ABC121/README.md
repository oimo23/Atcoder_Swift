### A-White Cells
```Swift
func whiteCells(H: Int, W: Int, h: Int, w: Int) -> Int {
  return (H * W) - (h * W) - ((w * H) - (h * w))
}

print(whiteCells(H: 3, W: 2, h: 2, w: 1))
```
